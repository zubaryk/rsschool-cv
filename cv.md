function mergeArrays(arr1, arr2) {

  const arr3 = [].concat(arr1, arr2);
  
  function compareNumbers(a, b) {
    return (a - b);
  }
  
  arr3.sort(compareNumbers);

  const newArr = Array.from(newSet);

}
